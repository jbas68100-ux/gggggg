-- البحث
local function FindPlayer(name)
    name = name:lower()
    for _,p in ipairs(Players:GetPlayers()) do
        if p ~= LocalPlayer and p.Name:lower():find(name) then
            return p
        end
    end
end

-- لبس
local function Wear(id)
    if tonumber(id) then
        Rep.Remotes.Wear:InvokeServer(id)
    end
end

-- تصفير
local function ResetBody()
    Rep.Remotes.ChangeCharacterBody:InvokeServer({0,0,0,0,0,0},"AllBlocky")
end

-- نسخ
local function Copy(desc)
    Rep.Remotes.ChangeCharacterBody:InvokeServer({
        desc.Torso,
        desc.RightArm,
        desc.LeftArm,
        desc.RightLeg,
        desc.LeftLeg,
        desc.Head
    },"LOC4THub")

    task.wait(0.2)

    Wear(desc.Shirt)
    Wear(desc.Pants)
    Wear(desc.Face)

    for _,v in ipairs(desc:GetAccessories(true)) do
        Wear(v.AssetId)
        task.wait(0.1)
    end
end

local function countMatch(a,b)
    local c=0
    for id in pairs(a) do
        if b[id] then c+=1 end
    end
    return c
end

local function tweenTo(plr)
    local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
    local thrp = plr.Character and plr.Character:FindFirstChild("HumanoidRootPart")
    if hrp and thrp then
        TweenService:Create(hrp,TweenInfo.new(0.2),{CFrame=thrp.CFrame}):Play()
    end
end

local function watch(plr)
    if plr == player then return end
    task.spawn(function()
        while enabledCopy and plr.Parent do
            local lhum = player.Character and player.Character:FindFirstChildOfClass("Humanoid")
            local thum = plr.Character and plr.Character:FindFirstChildOfClass("Humanoid")
            if lhum and thum then
                if countMatch(getAccSet(lhum:GetAppliedDescription()), getAccSet(thum:GetAppliedDescription())) >= 3 then
                    resetRemote:FireServer()
                    task.wait(0.1)
                    tweenTo(plr)
                end
            end
            task.wait(0.2)
        end
    end)
end

local function startBang(front)
    if not chasingplayer then return end

    local my = workspace:FindFirstChild(LocalPlayer.Name)
    local target = workspace:FindFirstChild(chasingplayer)

    if not (my and my:FindFirstChild("FollowCharacter") and target) then return end

    local follow = my.FollowCharacter
    local hrp = target:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    follow.Parent = workspace

    if getgenv().ChildLoop then
        getgenv().ChildLoop:Disconnect()
    end

    getgenv().ChildLoop = RunService.Heartbeat:Connect(function()
        if not (target and follow and follow:FindFirstChild("Torso")) then return end

        local basePos = front and (hrp.Position + hrp.CFrame.LookVector * 0.2)
                              or (hrp.Position - hrp.CFrame.LookVector * 0.2)

        local pos1 = basePos + (hrp.CFrame.LookVector * (front and -0.2 or -2.5))
        local pos2 = basePos + (hrp.CFrame.LookVector * (front and 2.5 or 0.2))

        local t = math.abs(math.sin(tick()*3))
        local newPos = pos1:Lerp(pos2, t)

        follow.Torso.BodyPosition.Position = newPos
        follow.Torso.BodyGyro.CFrame = CFrame.lookAt(follow.Torso.Position, hrp.Position)
    end)
end

local activeMode = nil
local currentTracks = {}


local function getHumanoid()
    local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    return char:WaitForChild("Humanoid")
end


local function stopAll()
    for _,track in pairs(currentTracks) do
        pcall(function()
            track:Stop()
            track:Destroy()
        end)
    end
    currentTracks = {}
    activeMode = nil

    pcall(function()
        Rep.Remotes.ResetCharacterAppearance:FireServer()
    end)
end


local function playAnim(id, speed)
    local hum = getHumanoid()

    local animator = hum:FindFirstChildOfClass("Animator")
    if not animator then
        animator = Instance.new("Animator", hum)
    end

    local anim = Instance.new("Animation")
    anim.AnimationId = "rbxassetid://"..id

    local track = animator:LoadAnimation(anim)
    track:Play()

    if speed then
        track:AdjustSpeed(speed)
    end

    table.insert(currentTracks, track)
    return track
end

    end)
end

local Tab = Window:MakeTab({
    Name = "Teleport",
    Icon = "rbxassetid://10734952036"
})

Tab:AddSection({
    Name = "⚡ Teleport System"
})

local TeleportTime = 0.1

local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local TweenService = game:GetService("TweenService")

local localPlayer = Players.LocalPlayer

local function resetCameraSubject()
	if workspace.CurrentCamera and localPlayer.Character then
		local humanoid = localPlayer.Character:FindFirstChildWhichIsA("Humanoid")
		if humanoid then
			workspace.CurrentCamera.CameraSubject = humanoid
		end
	end
end

local plr = Players.LocalPlayer
local character = plr.Character or plr.CharacterAdded:Wait()
local hrp = character:WaitForChild("HumanoidRootPart")

local toggle = false
local lastPos = hrp.Position
local targetPos = Vector3.new(0, -80000000, 0)
local db = false
local velConn

for _, gaze in ipairs(workspace:GetDescendants()) do
	if gaze:IsA("Part") and gaze.Name == "Gaze" and gaze.Transparency == 0.5 then
        gaze:Destroy()
	end
end

local function createTween(targetCFrame)
    local tweenInfo = TweenInfo.new(TeleportTime, Enum.EasingStyle.Linear, Enum.EasingDirection.Out)
    return TweenService:Create(hrp, tweenInfo, {CFrame = targetCFrame})
end

local function startVelLoop()
    velConn = RunService.Heartbeat:Connect(function()
        if toggle then
            hrp.Velocity = Vector3.new(0, 0, 0)
        end
    end)
end

local function stopVelLoop()
    if velConn then
        velConn:Disconnect()
        velConn = nil
    end
end

local function toggleTeleport()
    if db then return end

    if toggle then
        db = true
        toggle = false
        stopVelLoop()

        local tweenBack = createTween(CFrame.new(lastPos))
        tweenBack:Play()
        tweenBack.Completed:Wait()

        workspace.FallenPartsDestroyHeight = -500
        resetCameraSubject()

        for _, gaze in ipairs(workspace:GetDescendants()) do
		    if gaze:IsA("Part") and gaze.Name == "Gaze" then
                gaze:Destroy()
			end
        end

        db = false
    else
        db = true
        lastPos = hrp.Position

        local part = Instance.new("Part")
        part.Size = Vector3.new(4, 5, 4)
        part.Position = lastPos
        part.Anchored = true
        part.CanCollide = false
        part.Transparency = 0.5
        part.Name = "Gaze"
        part.Parent = workspace

        workspace.CurrentCamera.CameraSubject = part

        toggle = true
        task.wait()

        workspace.FallenPartsDestroyHeight = 0/0

        local tweenToTarget = createTween(CFrame.new(targetPos))
        tweenToTarget:Play()
        tweenToTarget.Completed:Wait()

        db = false
        startVelLoop()
    end
end

Tab:AddButton({
    Name = "تشغيل / ايقاف Teleport",
    Callback = function()
        toggleTeleport()
    end
})

            local fire = Instance.new("Fire")
            fire.Name = "KeroFire"
